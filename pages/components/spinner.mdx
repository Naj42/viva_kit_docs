### Example

Use spinners within buttons to provide visual feedback when an action is in progress or processing. You can swap the button text with spinner elements to indicate ongoing tasks.

<Tabs items={['Preview', 'Code']}>

<Tabs.Tab>
export const ButtonSpinnerGroup = () => {
  return (
    <div className="flex justify-center items-center gap-3 flex-wrap">
      <Button size="md" variant="outline"><Spinner size="sm" className="mr-2"/>Loading...</Button>
      <Button size="md" variant="solid"><Spinner size="sm" color="light" className="mr-2"/>Loading...</Button>
    </div>
  );
};

<ButtonSpinnerGroup />
</Tabs.Tab>

<Tabs.Tab>
```jsx copy
import React from 'react';
import { Spinner } from "@vivakits/react-components";

const App = () => (

return(

<div className="flex justify-center items-center gap-3 flex-wrap">
  <Button size="md" variant="outline">
    <Spinner size="sm" className="mr-2" />
    Loading...
  </Button>
  <Button size="md" variant="solid">
    <Spinner size="sm" color="light" className="mr-2" />
    Loading...
  </Button>
</div>
));

export default App;
```
