# Input

### Import

```jsx copy
import { Input } from "@vivakits/react-components";
```

### Size

The Input component comes with 4 different sizes that you can change using the `size` prop.

<Tabs items={['Preview', 'Code']}>

<Tabs.Tab>
export const InputSizeGroup = () => {
  return (
    <div className="flex items-end gap-3">
        <Input size="sm" label="Small" />
        <Input size="md" label="Medium" />
        <Input size="lg" label="Large" />
        <Input size="xl" label="Extra Large" />
    </div>
  )
}

<InputSizeGroup />
</Tabs.Tab>

  <Tabs.Tab>
```jsx copy
import { Input } from "@vivakits/react-components";

const App = () => {
  return (
    <div className="flex items-end gap-3">
        <Input size="sm" label="Small" />
        <Input size="md" label="Medium" />
        <Input size="lg" label="Large" />
        <Input size="xl" label="Extra Large" />
    </div>
  )
}

export default App;
  ```

### Icon

An icon can be appended or prepended using `icon` prop. The position of the icon can be specified by the `iconPosition` prop.

<Tabs items={['Preview', 'Code']}>

<Tabs.Tab>
export const InputIconGroup = () => {
  return (
    <div className="flex gap-3">
      <Input icon={<FaSearch />} placeholder="Search" />
      <Input iconPosition="end" icon={<FaSearch />} placeholder="Search" />
    </div>
  );
};

<InputIconGroup />
</Tabs.Tab>

<Tabs.Tab>
  ```jsx copy
import { Input } from "@vivakits/react-components";
import { FaSearch } from "react-icons/fa";

const App = () => {
    return (
      <div className="flex gap-3">
        <Input icon={<FaSearch />} placeholder="Search" />
        <Input iconPosition="end" icon={<FaSearch />} placeholder="Search" />
      </div>
    )
}

export default App;
  ```
</Tabs.Tab>

</Tabs>

### Error

Conditional error handling for input can be done via `hasError` prop. Error message can also be modified via `errorMessage` prop.

<Tabs items={['Preview', 'Code']}>

<Tabs.Tab>
export const InputErrorGroup = () => {
  return (
    <div className="flex gap-3">
      <Input icon={<FaExclamation />} placeholder="Default Error" hasError={true} />
      <Input icon={<FaExclamation />} placeholder="Custom Error" hasError={true} errorMessage="Custom error message" />
    </div>
  )
}

<InputErrorGroup />
</Tabs.Tab>

<Tabs.Tab>
  ```jsx copy
import { Input } from "@vivakits/react-components";
import { FaExclamation } from "react-icons/fa";

const App = () => {
    return (
      <div className="flex gap-3">
        <Input icon={<FaExclamation />} placeholder="Default Error" hasError={true} />
        <Input icon={<FaExclamation />} placeholder="Custom Error" hasError={true} errorMessage="Custom error message" />
      </div>
    )
}

export default App;
  ```
</Tabs.Tab>
