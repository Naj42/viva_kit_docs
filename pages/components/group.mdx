export const GroupColorGroup = () => {
  const colors = [
    {color: "primary", fontColor: "light"}, 
    {color: "secondary", fontColor: "light"},
    {color: "success", fontColor: "light"},
    {color: "warning", fontColor: "light"},
    {color: "danger", fontColor: "light"},
    {color: "dark", fontColor: "light"},
    {color: "light", fontColor: "dark"},

  return (
    <div className="flex flex-wrap items-start gap-2">

        colors.map(({color, fontColor}, index) => (
          <VKGroup
            color={color}
            key={index}
