### Highlight string

Highlight component render the main string as children and pass the string you want to highlight using the `highlightText` prop

<Tabs items={['Preview', 'Code']}>
<Tabs.Tab>
export const SingleString= () => {
  return (
    <div>
    <VKHighlight highlightText="highlight your selection">
        With the Highlight component, you can highlight your selection.
    </VKHighlight>
    </div>
  );
};

<SingleString />
</Tabs.Tab>

### Rounded

You can enhance highlighted strings by customizing their corner radius using `rounded` prop.  

<Tabs items={['Preview', 'Code']}>
<Tabs.Tab>
export const VarientsRadius= () => {
  const [rounded, setRounded] = useState([]);
  const handleChange = (e) =>{
    setRounded(e.target.value);
  }
  return (
    <div className="w-full flex flex-row-reverse justify-between items-center">
      <div className="flex flex-col gap-2">
        <label for="radius">Choose radius</label>
        <select name="radius" id="radius" value={rounded} onChange={handleChange} className="px-3 py-1">
          <option value="default">default</option>
          <option value="none">none</option>
          <option value="xs">xs</option>
          <option value="sm">sm</option>
          <option value="md">md</option>
          <option value="lg">lg</option>
          <option value="full">full</option>
        </select>
      </div>
      <VKHighlight highlightText="corner radius" rounded={`${rounded}`} background="secondary" className="px-3 py-1">
          You can change corner radius of your selection.
      </VKHighlight>
    </div>
  );
};

<VarientsRadius />
</Tabs.Tab>

### Styles

You can make more cutomization to meet your design, using `style` or `className` prop.

<Tabs items={['Preview', 'Code']}>
<Tabs.Tab>
export const AddingStyles= () => {
  return (
    <div className="flex flex-col gap-2">
        <VKHighlight highlightText="style" style={{background:"pink", color: "black", padding:"5px"}}>
            Customize selection using style. 
        </VKHighlight>
        <VKHighlight highlightText="className" className="px-3 py-1 bg-yellow-100 text-black rounded-xl">
            Customize selection using className. 
        </VKHighlight>
    </div>
  );
};

<AddingStyles />
