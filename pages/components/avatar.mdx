# Avatar

### Import

```jsx copy
import { VKAvatar } from "@vivakits/react-components";
```

### Variant

The Avatar component comes with 2 different variants that you can change using the `variant` prop. If no variant is provided, a default icon is shown.

<Tabs items={['Preview', 'Code']}>

  <Tabs.Tab>
export const AvatarVariantGroup = () => {

return (

<div className="flex justify-center items-center gap-6">
  <VKAvatar
    variant="image"
    size="md"
    src="https://i.pravatar.cc/150?u=a04258114e29026302d"
  />
  <VKAvatar variant="text" size="md" textValue="Viva Kit" />
  <VKAvatar size="md" />
</div>
) };

    <AvatarVariantGroup />

</Tabs.Tab>

  <Tabs.Tab>
```jsx copy
import { VKAvatar } from "@vivakits/react-components";

const App = () => {
  return (
    <div className="flex justify-center items-center gap-6">
        <VKAvatar
          variant="image"
          size="md"
          src="https://i.pravatar.cc/150?u=a04258114e29026302d"
        />
        <VKAvatar variant="text" size="md" textValue="Viva Kit" />
        <VKAvatar size="md" />
    </div>
  );
};

export default App;

````
