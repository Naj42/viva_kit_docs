# Action Icon

### Import

```jsx copy
import { VKActionIcon } from "@vivakits/react-components";
```

### Variant

The ActionIcon component comes with 3 different variants that you can change using the `variant` prop.

<Tabs items={['Preview', 'Code']}>

  <Tabs.Tab>
export const ActionIconVariantGroup = () => {
  return (
    <div className="flex justify-center items-center gap-3">
        <VKActionIcon icon={<SearchIcon/>} size="md" rounded="md" variant="solid"/>
        <VKActionIcon icon={<SearchIcon/>} size="md" rounded="md" variant="outline"/>
        <VKActionIcon icon={<SearchIcon/>} size="md" rounded="md" variant="light"/>
    </div>
  );
};

<ActionIconVariantGroup />
  </Tabs.Tab>

  <Tabs.Tab>
  ```jsx copy
import { VKActionIcon } from "@vivakits/react-components";

const App = () => {
    return (
      <div className="flex justify-center items-center gap-3">

    ); 
  };

export default App;
  ```
