### Radio Group

The Radio Button can be used inside `VKGroup` component to build radio button group.

<Tabs items={['Preview', 'Code']}>

<Tabs.Tab>
export const RadioButtonGroup = () => {
  const [selectedHorizontal, setSelectedHorizontal] = useState("");
  const [selectedVertical, setSelectedVertical] = useState("");
  const onChangeHorizontalHandler = (event) => {
    setSelectedHorizontal(event.target.value);
  }
  const onChangeVerticalHandler = (event) => {
    setSelectedVertical(event.target.value);
  }

  return (
    <div className="flex flex-col items-start gap-3 flex-wrap">
      <div className="flex flex-col mb-4">
        <VKTitle className="mb-2" variant="h6">Horizontal Radio Button Group</VKTitle>
        <VKGroup color="none" border="none" orientation="horizontal" childrenProps={{wrapperClassName: "mr-2"}} noSeparator>
          <VKRadioButton
            value={"accounting"}
            selectedValue={selectedHorizontal}
            onChange={onChangeHorizontalHandler}
          >
            Accounting
          </VKRadioButton>
          <VKRadioButton
            value={"marketing"}
            selectedValue={selectedHorizontal}
            onChange={onChangeHorizontalHandler}
          >
            Marketing
          </VKRadioButton>
          <VKRadioButton
            value={"production"}
            selectedValue={selectedHorizontal}
            onChange={onChangeHorizontalHandler}
          >
            Production
          </VKRadioButton>
          <VKRadioButton
            value={"research"}
            selectedValue={selectedHorizontal}
            onChange={onChangeHorizontalHandler}
          >
            Research
          </VKRadioButton>
        </VKGroup>
      </div>
      <div className="flex flex-col">
        <VKTitle className="mb-2" variant="h6">Vertical Radio Button Group</VKTitle>
        <VKGroup color="none" border="none" orientation="vertical" childrenProps={{wrapperClassName: "mr-2 py-0.5"}} noSeparator>
          <VKRadioButton
            value={"accounting"}
            selectedValue={selectedVertical}
            onChange={onChangeVerticalHandler}
          >
            Accounting
          </VKRadioButton>

//Horizontal Radio ButtonGroup

export function App() {
  const [selected, setSelected] = useState("");

    <VKGroup color="none" border="none" orientation="horizontal" childrenProps={{wrapperClassName: "mr-2"}} noSeparator>
      <VKRadioButton
        value={"accounting"}
        selectedValue={selected}
        onChange={onChangeHandler}
      >
        Accounting
      </VKRadioButton>
