# Badge

### Variants

The Badge component comes with 2 different variants that can be changed using the `variant` prop.

<Tabs items={['Preview', 'Code']}>

  <Tabs.Tab>
export const BadgeVariantGroup = () => {
  return (
    
    <div className="flex items-center gap-3">
    <VKBadge variant={"circular"}> 
      <img className="h-16 w-16" src="https://pic.onlinewebfonts.com/thumbnails/icons_390874.svg"/></VKBadge>
    <br />
    <VKBadge variant={"square"} ><img className="h-16 w-16" src="https://freesvg.org/img/1475409574.png"/></VKBadge>
    </div>

);
};

<BadgeVariantGroup />
  </Tabs.Tab>
