# Badge

### Variants

The Badge component comes with 2 different variants that can be changed using the `variant` prop.

<Tabs items={['Preview', 'Code']}>

  <Tabs.Tab>
export const BadgeVariantGroup = () => {
  return (
    
    <div className="flex items-center gap-3">
    <VKBadge variant={"circular"}  >Contact</VKBadge>
    <br />
    <VKBadge variant={"square"} >Contact</VKBadge>
    </div>

);
};

<BadgeVariantGroup />
  </Tabs.Tab>
