# Checkbox

### Import

```jsx copy
import { Checkbox } from "@vivakits/react-components";
```

### Size

The Checkbox component comes with 3 different sizes that you can change it using the `size` prop.

<Tabs items={['Preview', 'Code']}>

<Tabs.Tab>
export const CheckboxSizeGroup = () => {
  return (
    <div className="flex gap-3">
        <Checkbox size="sm">Small</Checkbox>
        <Checkbox size="md">Medium</Checkbox>
        <Checkbox size="lg">Large</Checkbox>
    </div>
  );
};

<CheckboxSizeGroup />
</Tabs.Tab>

  <Tabs.Tab>
```jsx copy
import { Checkbox } from "@vivakits/react-components";

const App = () => {
  return (
    <div className="flex gap-3">
        <Checkbox size="sm">Small</Checkbox>
        <Checkbox size="md">Medium</Checkbox>
        <Checkbox size="lg">Large</Checkbox>
    </div>
  )
}

export default App;
```

### Color

The Checkbox component comes with 7 different colors that you can change it using the `color` prop. It has both solid and outline variant

<Tabs items={['Preview', 'Code']}>

<Tabs.Tab>
export const CheckboxColorGroup = () => {
  return (
    <div className="flex gap-3 flex-wrap">
        <Checkbox color="primary">Primary</Checkbox>
        <Checkbox color="secondary">Secondary</Checkbox>
        <Checkbox color="success">Success</Checkbox>
        <Checkbox color="warning">Warning</Checkbox>
        <Checkbox color="danger">Danger</Checkbox>
        <Checkbox color="dark">Dark</Checkbox>
        <Checkbox color="accent">Accent</Checkbox>
    </div>
  );
};

<CheckboxColorGroup />
  </Tabs.Tab>

### Rounded

The shape of the Checkbox component can be updated using `rounded` prop

<Tabs items={['Preview', 'Code']}>

<Tabs.Tab>
export const CheckboxRoundedGroup = () => {
  return (
    <div className="flex gap-3">
        <Checkbox rounded="none">None</Checkbox>
        <Checkbox rounded="default">Default</Checkbox>
        <Checkbox rounded="sm">Small</Checkbox>
        <Checkbox rounded="md">Medium</Checkbox>
        <Checkbox rounded="lg">Large</Checkbox>
        <Checkbox rounded="full">Full</Checkbox>
    </div>
  )
}

<CheckboxRoundedGroup />
</Tabs.Tab>

<Tabs.Tab>

```jsx copy
 const App = () => {

        <Checkbox rounded="sm">Small</Checkbox>
        <Checkbox rounded="md">Medium</Checkbox>
        <Checkbox rounded="lg">Large</Checkbox>
        <Checkbox rounded="full">Full</Checkbox>
    </div>
  )
}

export default App;
```
