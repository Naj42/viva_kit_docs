# Popconfirm

### Title

The Popconfirm component can display any title passed to it using the `title` prop.

<Tabs items={['Preview', 'Code']}>

<Tabs.Tab>
export const PopconfirmTitleGroup = () => {
  return (
    <div className="flex space-x-1">
      <Popconfirm
              position="topLeft"
              title="This is a popconfirm"
              yesButton={
                <button
                  className="bg-white text-black rounded-lg p-1"
                  style={{ minWidth: "30%" }}
                >
                  Yes
                </button>
              }
              noButton={
                <button
                  className="border border-white rounded-lg p-1 w-1/3"
                  style={{ minWidth: "30%" }}
                >
                  No
                </button>
              }
            >
              <Button>Click me</Button>
            </Popconfirm>
    </div>
  );
};

<PopconfirmTitleGroup />
</Tabs.Tab>

</Tabs>
