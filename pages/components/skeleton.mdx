# Skeleton

### Variants

The Skeleton component comes with 3 different variants that you can change it using the `variant` prop.

<Tabs items={['Preview', 'Code']}>

  <Tabs.Tab>
export const SkeletonVarientGroup = () => {
  return (
    <div className="flex items-center gap-3">
        <Skeleton variant="circular" width={40} height={40} />
        <Skeleton variant="rectangular" height={40} />
        <Skeleton variant="text" height={40} />
    </div>
  );
};

<SkeletonVarientGroup />
  </Tabs.Tab>

  <Tabs.Tab>
```jsx copy
<div className="flex items-center gap-3">
  <Skeleton variant="circular" width="40px" height="40px" />
  <Skeleton variant="rectangular" height="40px" />
  <Skeleton variant="text" height="40px" />
</div>
  ```
  </Tabs.Tab>
</Tabs>

### Group of Skeleton
You can use props named `numOfRows` and `numOfColumns` to generate multiple rows or columns of Skeleton.

<Tabs items={['Preview', 'Code']}>

  <Tabs.Tab>
export const SkeletonRowColumnGroup = () => {
  return (
    <div className="flex items-center gap-5">
      <Skeleton numOfColumns={2}/>
      <Skeleton numOfRows={2}/>
      <Skeleton numOfColumns={2} numOfRows={2}/>
      <Skeleton numOfColumns={2} numOfRows={3}/>
    </div>
  );
};

<SkeletonRowColumnGroup />
  </Tabs.Tab>

### Table
Sample Table implementation using different varients of Skeleton components.

<Tabs items={['Preview', 'Code']}>

  <Tabs.Tab>
export const SkeletonTable = () => {
  const tableHeaders = ['Name', 'Position', 'Age', 'Start Date', 'Salary']
  return (
    <table className="w-fit p-5 rounded-md border mx-auto">
      <thead>
        <tr style={{ borderTop: '0px' }}>
          {tableHeaders.map((header, index) => (
            <th className='border-none font-bold' key={index}>
              {header}
            </th>
          ))}
        </tr>
      </thead>
      <tbody>
        {[...Array(10)].map((_, rowIndex) => (
          <tr style={{backgroundColor: 'white'}} key={rowIndex} className=''>
            {[...Array(5)].map((_, colIndex) => (
              <td className='border-none bg-none' key={`${rowIndex}-${colIndex}`}>
                <Skeleton variant="text" height={20} rounded="6px"/>
              </td>
            ))}
          </tr>
        ))}
      </tbody>
    </table>
  );
};

<SkeletonTable />
  </Tabs.Tab>
