### Variants

The Notification component comes with 4 different variants that you can change using the `variant` prop inside the `notify` function.

<Tabs items={['Preview', 'Code']}>

<Tabs.Tab>
export const NotificationsVariantGroup = () => {
    const { notify } = useVKNotifications();
  return (
    <>
        <VKNotifications />
        <div className="flex items-center gap-3">
            <Button
                size="md"
                rounded="lg"
                onClick={() => {
                    notify({
                        variant: "default",
                        title: "Default",
                        description: "Default Notification",
                        autoClose: true,
                        duration: 3000

                Default
