# Title

### Import

```jsx copy
import { VKTitle } from "@vivakits/react-components";
```

### Variants

The Title component comes with 7 different variants that you can change using the `variant` prop.

<Tabs items={['Preview', 'Code']}>

  <Tabs.Tab>
export const TitleVariantGroup = () => {
    const variants = [ "h1" ,"h2", "h3", "h4", "h5", "h6", "paragraph" ];

return (

<div className="flex gap-6">
  {variants.map((variant) => (
    <VKTitle variant={variant}>{variant}</VKTitle>
  ))}
</div>
); };

    <TitleVariantGroup />

</Tabs.Tab>

  <Tabs.Tab>
```jsx copy

    <div className="flex gap-6">
        <VKTitle variant="h1">h1</VKTitle>
        <VKTitle variant="h2">h2</VKTitle>
        <VKTitle variant="h3">h3</VKTitle>
        <VKTitle variant="h4">h4</VKTitle>
        <VKTitle variant="h5">h5</VKTitle>
        <VKTitle variant="h6">h6</VKTitle>
        <VKTitle variant="paragraph">paragraph</VKTitle>
    </div>

````
</Tabs.Tab>
