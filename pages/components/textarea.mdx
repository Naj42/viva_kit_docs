# Textarea

### Variant

The Textarea component comes with 2 different variants that you can change it using the `variant` prop.

<Tabs items={['Preview', 'Code']}>

  <Tabs.Tab>
export const TextareaVariantGroup = () => {
  return (
    <div className="flex items-center gap-3">
        <Textarea label="Outlined:" variant="outline"  placeholder="Enter text"/>
        <Textarea label="Solid:" variant="solid"  placeholder="Enter text" className="!bg-gray-200" />        
    </div>
  );
};

<TextareaVariantGroup />
  </Tabs.Tab>

  <Tabs.Tab>
```jsx copy
import React from 'react';
import { Textarea } from "@vivakits/react-components";

const App = () => {
  return (
    <div className="flex items-center gap-3">
      <Textarea label="Outlined Textarea variant" variant="outline" />
      <Textarea
        label="Solid Textarea variant"
        variant="solid"
        className="!bg-gray-300"
      />
    </div>
  )
};

export default App;
```
  </Tabs.Tab>
</Tabs>

### Icon

An icon can be append or prepend using `icon` prop. The position of the icon can be specified by the `iconPosition` prop.

<Tabs items={['Preview', 'Code']}>

<Tabs.Tab>
export const TextareaIconGroup = () => {
  return (
    <div className="flex items-end gap-3">
      <Textarea icon={<FaSearch />} placeholder="Search" />
      <Textarea iconPosition="end" icon={<FaSearch />} placeholder="Search" />
    </div>
  );
};

<TextareaIconGroup />
</Tabs.Tab>

<Tabs.Tab>
```jsx copy
import React from 'react';
import { Textarea } from "@vivakits/react-components";

const App = () => {
  return (
    <div className="flex items-end gap-3">
      <Textarea icon={<FaSearch />} placeholder="Search" />
      <Textarea iconPosition="end" icon={<FaSearch />} placeholder="Search" />
    </div>
  )
};

export default App;
```
</Tabs.Tab>

### Error

Conditional error handling for Textarea can be done via `hasError` props. Error message can also be modified via `errorMessage` props.

<Tabs items={['Preview', 'Code']}>

<Tabs.Tab>
export const TextareaErrorGroup = () => {
  return (
    <div className="flex items-end gap-3">
      <Textarea icon={<FaEdit />} placeholder="Error variant" hasError={true} />
      <Textarea icon={<FaEdit />} placeholder="Error variant" hasError={true} errorMessage="Custom error message" />
    </div>
  );
};

<TextareaErrorGroup />
</Tabs.Tab>

<Tabs.Tab>
```jsx copy
import React from 'react';
import { Textarea } from "@vivakits/react-components";

const App = () => {
  return (
    <div className="flex items-end gap-3">
      <Textarea icon={<FaEdit />} placeholder="Error variant" hasError={true} />
      <Textarea
        icon={<FaEdit />}
        placeholder="Error variant"
        hasError={true}
        errorMessage="Custom error message"
      />
    </div>
  )
};

export default App;
```
</Tabs.Tab>
</Tabs>
